"""
Schema per le risposte del modulo setup.
"""

from pydantic import BaseModel, ConfigDict, Field

from app.modules.products.schemas import ProductBulkLoadResponse




[docs]
class SetupProductsResponse(BaseModel):
    """Risposta dell'endpoint di setup prodotti di test."""

    message: str = Field(..., description="Messaggio di conferma")
    result: ProductBulkLoadResponse = Field(
        ..., description="Risultato dell'operazione di caricamento anagrafica"
    )

    model_config = ConfigDict(
        json_schema_extra={
            "example": {
                "message": "Setup completato delegando a carica_anagrafica",
                "result": {
                    "status": "success",
                    "messaggio": "Operazione completata per DefaultStore",
                    "riepilogo": {"nuovi_inseriti": 5, "gia_presenti": 2},
                },
            }
        }
    )

class SetupKitchenResponse(BaseModel):
    """Risposta dell'endpoint di setup dati cucina."""

    message: str = Field(..., description="Messaggio di conferma")
    created_recipes: list[str] = Field(..., description="Nomi delle ricette create")

    model_config = ConfigDict(
        json_schema_extra={
            "example": {
                "message": "Setup cucina completato",
                "created_recipes": ["Fajitas", "BBQ Jack Daniel's", "Salsa Puebla"],
            }
        }
    )
