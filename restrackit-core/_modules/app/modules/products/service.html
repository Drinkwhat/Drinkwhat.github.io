    def bulk_load_products(
        self, items: list[ProductCreate], store_id: int
    ) -> ProductBulkLoadResponse:
        """
        Carica o aggiorna massivamente l'anagrafica prodotti.

        Args:
            items: Lista di prodotti da inserire (o da saltare se già presenti).
            store_id: ID dello store (multi-tenancy).

        Returns:
            ProductBulkLoadResponse: Riepilogo dell'operazione (nuovi vs già presenti).

        Raises:
            DatabaseError: In caso di errore durante il commit complessivo.
        """
        nuovi = 0
        saltati = 0
        logger.info(f"📦 Starting bulk product upload for store {store_id} with {len(items)} items")

        for item in items:
            try:
                # Usiamo un savepoint per isolare ogni singolo inserimento
                with self.db.begin_nested():
                    # Cerchiamo se il prodotto esiste già per questo store
                    esistente = (
                        self.db.query(ProductMaster)
                        .filter(
                            ProductMaster.name == item.name.lower(),
                            ProductMaster.store_id == store_id,
                            ProductMaster.deleted_at.is_(None),
                        )
                        .first()
                    )

                    if not esistente:
                        new_p = ProductMaster(
                            name=item.name.lower(), store_id=store_id, category=item.category
                        )
                        self.db.add(new_p)
                        self.db.flush()  # Controlla i vincoli subito
                        nuovi += 1
                        logger.debug(f"✅ Product created: {item.name} (cat: {item.category})")
                    else:
                        # Se preferisci che aggiorni la categoria, decommenta la riga sotto:
                        # esistente.category = item.category
                        saltati += 1
                        logger.debug(f"↩️  Product already exists: {item.name}")
            except IntegrityError:
                saltati += 1
                logger.info(f"Duplicato saltato via IntegrityError: {item.name}")

        try:
            self.db.commit()  # Salva tutto quello che è andato a buon fine
            logger.info(f"✅ Bulk product upload completed: {nuovi} new, {saltati} skipped")
        except Exception as e:
            self.db.rollback()
            logger.error(f"❌ Database error during bulk commit: {e}")
            raise DatabaseError(ErrorCode.DB_CONNECTION_ERROR, original_exc=e) from e

        return ProductBulkLoadResponse(
            status="success",
            messaggio=f"Operazione completata per {store_id}",
            riepilogo={"nuovi_inseriti": nuovi, "gia_presenti": saltati},
        )

    def list_products(self, store_id: int) -> list[ProductResponse]:
        """Recupera i prodotti dello store autorizzato.

        Args:
            store_id: ID dello store (multi-tenancy).

        Returns:
            list[ProductResponse]: Prodotti attivi ordinati per nome.
        """
        prodotti = (
            self.db.query(ProductMaster)
            .filter(ProductMaster.store_id == store_id, ProductMaster.deleted_at.is_(None))
            .order_by(ProductMaster.name)
            .all()
        )

        logger.debug(f"📋 Retrieved {len(prodotti)} products for store {store_id}")
        return [ProductResponse.model_validate(p) for p in prodotti]

    def update_product(
        self, product_id: int, data: ProductUpdate, store_id: int
    ) -> ProductResponse:
        """Modifica puntuale di un prodotto (optimistic locking).

        Args:
            product_id: ID del prodotto.
            data: Payload di aggiornamento (include versione attesa).
            store_id: ID dello store (multi-tenancy).

        Returns:
            ProductResponse: Prodotto aggiornato.

        Raises:
            BusinessLogicError: Se prodotto non trovato, conflitto di versione o duplicato.
            DatabaseError: In caso di errore DB.
        """
        prodotto = (
            self.db.query(ProductMaster)
            .filter(
                ProductMaster.id == product_id,
                ProductMaster.store_id == store_id,
                ProductMaster.deleted_at.is_(None),
            )
            .first()
        )

        if not prodotto:
            logger.warning(f"⚠️  Product not found: id={product_id} for store {store_id}")
            raise BusinessLogicError(
                ErrorCode.BUS_PRODUCT_NOT_FOUND, params={"product": f"ID {product_id}"}
            )

        if prodotto.version != data.version:
            raise BusinessLogicError(
                ErrorCode.BUS_CONFLICT,
                params={
                    "resource": f"Product {product_id}",
                    "expected": data.version,
                    "current": prodotto.version,
                },
            )

        if data.name:
            prodotto.name = data.name  # type: ignore[assignment]
        if data.category:
            prodotto.category = data.category  # type: ignore[assignment]

        try:
            logger.debug(
                f"📝 Updating product: id={product_id}, name={data.name}, category={data.category}"
            )
            self.db.commit()
            self.db.refresh(prodotto)
            logger.info(f"✅ Product updated: id={product_id}")
            return ProductResponse.model_validate(prodotto)
        except StaleDataError as e:
            self.db.rollback()
            raise BusinessLogicError(
                ErrorCode.BUS_CONFLICT,
                params={
                    "resource": f"Product {product_id}",
                    "expected": data.version,
                    "current": "unknown",
                },
                original_exc=e,
            ) from e
        except IntegrityError as e:
            self.db.rollback()
            logger.warning(f"❌ Integrity error updating product {product_id}: duplicate name")
            raise BusinessLogicError(
                ErrorCode.BUS_ALREADY_EXISTS, params={"resource": f"Product name '{data.name}'"}
            ) from e
        except Exception as e:
            self.db.rollback()
            logger.error(f"❌ Error updating product {product_id}: {e!s}", exc_info=True)
            raise DatabaseError(ErrorCode.DB_QUERY_ERROR, original_exc=e) from e

    def delete_product(self, product_id: int, store_id: int, version: int) -> None:
        """Soft delete di un prodotto se non collegato a lotti attivi.

        Args:
            product_id: ID del prodotto.
            store_id: ID dello store (multi-tenancy).
            version: Versione attesa della risorsa.

        Raises:
            BusinessLogicError: Se prodotto non trovato, conflitto versione o prodotto in uso.
            DatabaseError: In caso di errore DB.
        """
        logger.info(f"🗑️  Attempting to delete product: id={product_id}, store={store_id}")

        prodotto = (
            self.db.query(ProductMaster)
            .filter(
                ProductMaster.id == product_id,
                ProductMaster.store_id == store_id,
                ProductMaster.deleted_at.is_(None),
            )
            .first()
        )

        if not prodotto:
            logger.warning(f"⚠️  Product not found for deletion: id={product_id}, store={store_id}")
            raise BusinessLogicError(
                ErrorCode.BUS_PRODUCT_NOT_FOUND, params={"product": f"ID {product_id}"}
            )

        if prodotto.version != version:
            raise BusinessLogicError(
                ErrorCode.BUS_CONFLICT,
                params={
                    "resource": f"Product {product_id}",
                    "expected": version,
                    "current": prodotto.version,
                },
            )

        has_batches = any(b.deleted_at is None for b in prodotto.batches)
        if has_batches:
            raise BusinessLogicError(
                ErrorCode.BUS_OPERATION_NOT_ALLOWED,
                params={"details": "Prodotto in uso in magazzino."},
            )

        try:
            logger.debug(f"📝 Soft deleting product: id={product_id}, name={prodotto.name}")
            prodotto.deleted_at = datetime.now(UTC)
            self.db.commit()
            logger.info(f"✅ Product soft deleted: id={product_id}, name={prodotto.name}")
        except StaleDataError as e:
            self.db.rollback()
            raise BusinessLogicError(
                ErrorCode.BUS_CONFLICT,
                params={
                    "resource": f"Product {product_id}",
                    "expected": version,
                    "current": "unknown",
                },
                original_exc=e,
            ) from e
        except Exception as e:
            self.db.rollback()
            logger.error(f"❌ Error deleting product {product_id}: {e!s}", exc_info=True)
            raise DatabaseError(
                ErrorCode.DB_QUERY_ERROR, params={"details": str(e)}, original_exc=e
            ) from e

    def get_product(self, product_id: int, store_id: int) -> ProductResponse:
        """Recupera un prodotto attivo (non soft-deleted) per ID e store.

        Args:
            product_id: ID del prodotto.
            store_id: ID dello store (multi-tenancy).

        Returns:
            ProductResponse: DTO del prodotto.

        Raises:
            BusinessLogicError: Se il prodotto non esiste nello store o è soft-deleted.
        """
        prodotto = (
            self.db.query(ProductMaster)
            .filter(
                ProductMaster.id == product_id,
                ProductMaster.store_id == store_id,
                ProductMaster.deleted_at.is_(None),
            )
            .first()
        )
        if not prodotto:
            raise BusinessLogicError(
                ErrorCode.BUS_PRODUCT_NOT_FOUND, params={"product": f"ID {product_id}"}
            )
        return ProductResponse.model_validate(prodotto)
