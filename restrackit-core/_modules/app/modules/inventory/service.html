    async def analyze_image(self, file_bytes: bytes, store_id: int) -> BatchAnalyzeResponse:
        """
        Analyze an image to extract product details using OCR and AI.
        Executes blocking calls in a thread pool to avoid blocking the event loop.

        Args:
            file_bytes: Raw bytes of the uploaded image.
            store_id: Store identifier used to scope product matching.

        Returns:
            BatchAnalyzeResponse: Parsed product data, optional warning, and OCR snippet.

        Raises:
            ValidationError: If the file is empty.
            ExternalServiceError: If OCR fails.
            BusinessLogicError: If AI parsing fails.
            SystemError: For unexpected errors.
        """
        if not file_bytes:
            logger.warning("⚠️  Empty file upload attempt")
            raise ValidationError(ErrorCode.VAL_EMPTY_FILE)

        try:
            # 1. OCR
            try:
                logger.debug("📷 Running OCR on image...")
                # Run sync OCR in threadpool
                raw_text = await run_in_threadpool(self.ocr_service.extract_text, file_bytes)
                logger.debug(f"✅ OCR completed, extracted text length: {len(raw_text)} chars")
            except AppError:
                raise
            except Exception as e:
                logger.error(f"❌ Vision OCR failure: {e}", exc_info=True)
                raise ExternalServiceError(
                    ErrorCode.EXT_VISION_ERROR, params={"details": str(e)}, original_exc=e
                ) from e

            # 2. Agent
            try:
                logger.debug("🤖 Running AI agent to parse OCR text...")
                # Run sync Agent in threadpool
                # We pass str(store_id) because the agent expects a string for the tool
                data = await run_in_threadpool(
                    self.agent_service.analyze_ocr_text, raw_text, str(store_id)
                )
                logger.debug(
                    f"✅ AI agent parsing completed: product={data.get('prodotto')}, "
                    f"confidence={data.get('confidence')}"
                )
            except AppError:
                raise
            except Exception as e:
                logger.error(f"❌ Parsing agent failure: {e}", exc_info=True)
                raise BusinessLogicError(ErrorCode.BUS_PARSING_ERROR, original_exc=e) from e

            # 3. Human-in-the-Loop Warning
            warning = None
            if data.get("prodotto") in ["NON_IDENTIFICATO", "ERRORE_AI", None]:
                warning = "Prodotto non riconosciuto in anagrafica. Selezionalo manualmente."
                logger.warning(
                    f"⚠️  AI unable to identify product for {store_id} | "
                    f"OCR snippet: {raw_text[:50]}"
                )
            else:
                logger.info(
                    f"✅ Product identified: {data.get('prodotto')} "
                    f"(confidence: {data.get('confidence')})"
                )

            return BatchAnalyzeResponse(
                status="revision", data=data, warning=warning, raw_ocr_debug=raw_text[:100]
            )
        except AppError:
            raise
        except Exception as e:
            logger.error(f"❌ Error in analyze_image: {e!s}", exc_info=True)
            raise SystemError(
                ErrorCode.SYS_INTERNAL_ERROR, params={"request_id": "unknown"}, original_exc=e
            ) from e

    def confirm_batch(self, data: BatchConfirm, store_id: int, operator_id: str) -> BatchResponse:
        """
        Confirm and save a validated batch to the database.

        Args:
            data: Payload confirmed by the user after review.
            store_id: Store identifier (multi-tenancy).
            operator_id: Identifier of the operator performing the action.

        Returns:
            BatchResponse: Saved batch information.

        Raises:
            BusinessLogicError: If the product does not exist in the store.
            DatabaseError: If database constraints are violated.
            SystemError: For unexpected errors.
        """
        logger.info(
            f"💾 Confirming batch load: product={data.product_name}, "
            f"lot={data.lot_code}, store={store_id}"
        )

        try:
            # Validate Product
            product_master = (
                self.db.query(ProductMaster)
                .filter_by(name=data.product_name, store_id=store_id)
                .first()
            )

            if not product_master:
                raise BusinessLogicError(
                    ErrorCode.BUS_PRODUCT_NOT_FOUND, params={"product": data.product_name}
                )
            if product_master.deleted_at is not None:
                raise BusinessLogicError(
                    ErrorCode.BUS_PRODUCT_NOT_FOUND, params={"product": data.product_name}
                )

            new_batch = Batch(
                store_id=store_id,
                operator_id=operator_id,
                product_master_id=product_master.id,
                lot_code=data.lot_code,
                lot_code_ai=data.lotto_proposto_ai,
                expiry_date=data.expiry_date,
                confidence=data.confidence,
                is_empty=False,
            )

            try:
                self.db.add(new_batch)
                self.db.commit()
                self.db.refresh(new_batch)
            except IntegrityError as e:
                self.db.rollback()
                logger.error(f"Errore DB in conferma: {e}")
                raise DatabaseError(
                    ErrorCode.DB_INTEGRITY_ERROR, params={"details": str(e)}, original_exc=e
                ) from e

            logger.info(
                f"✅ Batch {new_batch.id} saved | operator={operator_id}, "
                f"store={store_id}, product={product_master.name}"
            )

            return BatchResponse(
                status="success",
                message=f"Carico validato da {operator_id}",
                batch_id=new_batch.id,
                product_name=product_master.name,
                lot_code=new_batch.lot_code,
                lot_code_ai=new_batch.lot_code_ai,
                expiry_date=new_batch.expiry_date,
                store_id=new_batch.store_id,
                operator_id=new_batch.operator_id,
                confidence=new_batch.confidence,
                created_at=new_batch.created_at,
                updated_at=new_batch.updated_at,
                deleted_at=new_batch.deleted_at,
                version=new_batch.version,
            )
        except AppError:
            raise
        except Exception as e:
            self.db.rollback()
            logger.error(f"❌ Database error saving batch: {e!s}", exc_info=True)
            raise SystemError(
                ErrorCode.SYS_INTERNAL_ERROR, params={"request_id": "unknown"}, original_exc=e
            ) from e

    def update_batch(self, batch_id: int, data: BatchUpdate, store_id: int) -> None:
        """
        Update an existing batch.

        Args:
            batch_id: Batch identifier.
            data: Update payload (includes expected version).
            store_id: Store identifier (multi-tenancy).

        Raises:
            BusinessLogicError: If the batch does not exist or version conflict occurs.
            DatabaseError: If the database update fails.
        """
        lotto = (
            self.db.query(Batch)
            .filter(Batch.id == batch_id, Batch.store_id == store_id, Batch.deleted_at.is_(None))
            .first()
        )

        if not lotto:
            raise BusinessLogicError(
                ErrorCode.BUS_RESOURCE_NOT_FOUND, params={"resource": f"Batch {batch_id}"}
            )

        if lotto.version != data.version:
            raise BusinessLogicError(
                ErrorCode.BUS_CONFLICT,
                params={
                    "resource": f"Batch {batch_id}",
                    "expected": data.version,
                    "current": lotto.version,
                },
            )

        pm = (
            self.db.query(ProductMaster)
            .filter(
                ProductMaster.name == data.product_name,
                ProductMaster.store_id == store_id,
                ProductMaster.deleted_at.is_(None),
            )
            .first()
        )

        if not pm:
            raise BusinessLogicError(
                ErrorCode.BUS_PRODUCT_NOT_FOUND, params={"product": data.product_name}
            )

        lotto.product_master_id = pm.id
        lotto.lot_code = data.lot_code
        lotto.expiry_date = data.expiry_date
        lotto.confidence = 1.0

        try:
            self.db.commit()
        except StaleDataError as e:
            self.db.rollback()
            raise BusinessLogicError(
                ErrorCode.BUS_CONFLICT,
                params={
                    "resource": f"Batch {batch_id}",
                    "expected": data.version,
                    "current": "unknown",
                },
                original_exc=e,
            ) from e
        except Exception as e:
            self.db.rollback()
            logger.error(f"Errore aggiornamento lotto: {e}")
            raise DatabaseError(
                ErrorCode.DB_INTEGRITY_ERROR, params={"details": str(e)}, original_exc=e
            ) from e

    def list_batches(
        self, filters: BatchListFilter, pagination: PaginationParams, store_id: int
    ) -> BatchListResponse:
        """
        List batches with filtering and pagination.

        Args:
            filters: Filtering options.
            pagination: Pagination parameters.
            store_id: Store identifier (multi-tenancy).

        Returns:
            BatchListResponse: Paginated list of batches.
        """
        q = self.db.query(Batch).filter(Batch.store_id == store_id, Batch.deleted_at.is_(None))

        if filters.product_name:
            q = q.join(ProductMaster).filter(ProductMaster.name == filters.product_name.lower())

        if filters.is_empty is not None:
            q = q.filter(Batch.is_empty == filters.is_empty)

        if filters.expired is not None:
            today = datetime.now(UTC).date()
            if filters.expired:
                q = q.filter(Batch.expiry_date < today)
            else:
                q = q.filter(Batch.expiry_date >= today)

        total = q.count()
        rows = (
            q.order_by(Batch.created_at.desc())
            .offset(pagination.offset)
            .limit(pagination.page_size)
            .all()
        )

        items = []
        for b in rows:
            pm_name = b.product_master.name if b.product_master else ""
            items.append(
                BatchResponse(
                    status="success",
                    message="Lotto registrato",
                    batch_id=b.id,
                    product_name=pm_name,
                    lot_code=b.lot_code,
                    lot_code_ai=b.lot_code_ai,
                    expiry_date=b.expiry_date,
                    store_id=b.store_id,
                    operator_id=b.operator_id,
                    confidence=b.confidence,
                    created_at=b.created_at,
                    updated_at=b.updated_at,
                    deleted_at=b.deleted_at,
                    version=b.version,
                )
            )

        return BatchListResponse(status="success", count=total, items=items)

    def delete_batch(self, batch_id: int, store_id: int, version: int) -> None:
        """Soft delete di un lotto (optimistic locking).

        Args:
            batch_id: ID del lotto.
            store_id: ID dello store (multi-tenancy).
            version: Versione attesa della risorsa.

        Raises:
            BusinessLogicError: Se lotto non trovato o conflitto di versione.
            DatabaseError: In caso di errore DB.
        """
        lotto = (
            self.db.query(Batch)
            .filter(Batch.id == batch_id, Batch.store_id == store_id, Batch.deleted_at.is_(None))
            .first()
        )

        if not lotto:
            raise BusinessLogicError(
                ErrorCode.BUS_RESOURCE_NOT_FOUND, params={"resource": f"Batch {batch_id}"}
            )

        if lotto.version != version:
            raise BusinessLogicError(
                ErrorCode.BUS_CONFLICT,
                params={
                    "resource": f"Batch {batch_id}",
                    "expected": version,
                    "current": lotto.version,
                },
            )

        try:
            lotto.deleted_at = datetime.now(UTC)
            self.db.commit()
        except StaleDataError as e:
            self.db.rollback()
            raise BusinessLogicError(
                ErrorCode.BUS_CONFLICT,
                params={"resource": f"Batch {batch_id}", "expected": version, "current": "unknown"},
                original_exc=e,
            ) from e
        except Exception as e:
            self.db.rollback()
            logger.error(f"Errore eliminazione lotto: {e}")
            raise DatabaseError(
                ErrorCode.DB_INTEGRITY_ERROR, params={"details": str(e)}, original_exc=e
            ) from e

    def get_batch(self, batch_id: int, store_id: int) -> BatchResponse:
        """Recupera un lotto (batch) attivo per ID e store.

        Args:
            batch_id: Identificativo del lotto.
            store_id: Identificativo dello store (multi-tenancy).

        Returns:
            BatchResponse: DTO del lotto.

        Raises:
            BusinessLogicError: Se il lotto non esiste nello store o è soft-deleted.
        """
        lotto = (
            self.db.query(Batch)
            .filter(Batch.id == batch_id, Batch.store_id == store_id, Batch.deleted_at.is_(None))
            .first()
        )
        if not lotto:
            raise BusinessLogicError(
                ErrorCode.BUS_RESOURCE_NOT_FOUND, params={"resource": f"Batch {batch_id}"}
            )
        pm_name = lotto.product_master.name if lotto.product_master else ""
        return BatchResponse(
            status="success",
            message="Lotto",
            batch_id=lotto.id,
            product_name=pm_name,
            lot_code=lotto.lot_code,
            lot_code_ai=lotto.lot_code_ai,
